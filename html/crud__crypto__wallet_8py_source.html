13@crypto_wallet_bp.route('/add_wallet', methods=['GET', 'POST'])
14@login_required
15def add_wallet():
16 """
17 Esta função lida com a adição de uma nova carteira de criptomoedas. Primeiro, valida os dados do formulário,
18 verifica se a carteira já existe para o usuário atual e, em seguida, adiciona a nova carteira ao banco de dados.
19
20 Parâmetros:
21 - formAddWallet: Uma instância da classe AddWalletForm, que contém os dados do formulário para adicionar uma carteira.
22
23 Retorna:
24 - Se os dados do formulário são válidos e a carteira não existe, uma nova carteira é adicionada ao banco de dados,
25 uma mensagem de flash de sucesso é exibida e o usuário é redirecionado para a página de carteiras.
26 - Se os dados do formulário são inválidos ou a carteira já existe, uma mensagem de flash de erro é exibida.
27 - Se ocorre uma exceção durante as operações de banco de dados, uma reversão é executada e uma mensagem de flash de erro é exibida.
28 """
29 formAddWallet = AddWalletForm() # Criação do formulário
30 if formAddWallet.validate_on_submit():
31 wallet_name = formAddWallet.wallet_name.data.strip().upper()
32 wallet_network = formAddWallet.wallet_network.data.strip().upper()
33
34 try:
35 # Verificar se a carteira já existe para o usuário atual
36 existing_wallet = Wallet.query.filter_by(wallet_user_id=current_user.user_id, wallet_name=wallet_name).first()
37 if existing_wallet:
38 flash(f'A Carteira {wallet_name} já está cadastrada.', 'alert-warning')
39 return redirect(url_for('views.wallets'))
40
41 # Criar e adicionar a nova carteira
42 wallet = Wallet(wallet_user_id=current_user.user_id, wallet_name=wallet_name, wallet_network=wallet_network)
43 db.session.add(wallet)
44 db.session.commit()
45 flash(f'A Carteira {wallet_name} foi adicionada com sucesso', 'alert-success')
46 return redirect(url_for('views.wallets'))
47 except Exception as e:
48 db.session.rollback()
49 flash(f'Erro ao tentar adicionar a carteira: {e}', 'alert-danger')
50
51 return render_template('operacoes/add_wallet.html', formAddWallet=formAddWallet)

55@crypto_wallet_bp.route('/delete_wallet', methods=['POST'])
56@login_required
57def delete_wallet():
58 """
59 Esta função lida com a exclusão de uma carteira de criptomoedas. Primeiro, verifica se o ID da carteira é fornecido,
60 em seguida, verifica se a carteira existe para o usuário atual. Se a carteira existir, verifica se a carteira
61 possui alguma transação associada. Se a carteira não possuir transações, ela é excluída do banco de dados.
62 Se a carteira possuir transações, seu status é definido como 'S'.
63
64 Parâmetros:
65 - wallet_id (str): O ID da carteira a ser excluída. Este ID é obtido da solicitação POST do formulário.
66
67 Retorna:
68 - Se o ID da carteira não for fornecido, uma mensagem flash é exibida indicando que o ID da carteira não foi fornecido.
69 O usuário é então redirecionado para a página de carteiras.
70 - Se a carteira não existir para o usuário atual, uma mensagem flash é exibida indicando que a carteira não foi encontrada.
71 O usuário é então redirecionado para a página de carteiras.
72 - Se a carteira não possuir transações associadas, ela é excluída do banco de dados e uma mensagem flash é exibida indicando
73 que a carteira foi excluída com sucesso. O usuário é então redirecionado para a página de carteiras.
74 - Se a carteira possuir transações associadas, seu status é definido como 'S' e uma mensagem flash é exibida indicando que
75 a carteira foi desativada devido às transações existentes. O usuário é então redirecionado para a página de carteiras.
76 - Se ocorrer uma exceção durante as operações de banco de dados, uma reversão é executada, e uma mensagem flash é exibida indicando
77 o erro que ocorreu. O usuário é então redirecionado para a página de carteiras.
78 """
79 wallet_id = request.form.get('wallet_id')
80 if wallet_id:
81 try:
82 wallet = Wallet.query.filter_by(wallet_id=wallet_id, wallet_user_id=current_user.user_id).first()
83 if not wallet:
84 flash('Carteira não encontrada.', 'alert-danger')
85 return redirect(url_for('views.wallets'))
86
87 wallet_in_transactions = Transaction.query.filter(
88 or_(
89 Transaction.payment_wallet_id == wallet_id,
90 Transaction.receiving_wallet_id == wallet_id
91 )
92 ).first()
93
94 if not wallet_in_transactions:
95 db.session.delete(wallet)
96 db.session.commit()
97 flash('Carteira excluída com sucesso.', 'alert-success')
98 else:
99 wallet.wallet_status = 'S'
100 db.session.commit()
101 flash('Carteira desativada pois já teve transações com ela.', 'alert-success')
102 except Exception as e:
103 db.session.rollback()
104 flash(f'Erro ao tentar desativar a carteira: {e}', 'alert-danger')
105 else:
106 flash("ID da carteira não fornecido", 'alert-danger')
107
108 return redirect(url_for('views.wallets'))
